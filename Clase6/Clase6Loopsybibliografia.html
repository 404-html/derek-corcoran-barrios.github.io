Empezemos nuestro loop

Madrid <- map(Archivos, read_csv)
  • Si le agregamos argumentos a una función agregamos un ~ antes de la función y un .x en vez de los datos
library(lubridate)
Madrid <- map(Archivos, read_csv) %>% map(~filter(.x, station %in% c(28079036, 
    28079008, 28079058, 28079060, 28079038)))
  • A partir de esto podemos avanzar mucho
Madrid <- map(Archivos, read_csv) %>% map(~filter(.x, station %in% c(28079036, 
    28079008, 28079058, 28079060, 28079038))) %>% map(~mutate(.x, month = month(date), 
    year = year(date))) %>% map(~left_join(.x, stations)) %>% map(~group_by(.x, 
    month, name, year)) %>% map(~summarise(.x, NO_2 = mean(NO_2, na.rm = TRUE), 
    n = n())) %>% map(~filter(.x, n > 500))
